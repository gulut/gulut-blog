// 对当前用户禁用
adb shell pm disable-user AppPackageName
// 对本机禁用
adb shell pm disable AppPackageName
// 启用应用
adb shell pm enable AppPackageName
// 清除应用数据
adb shell pm clear AppPackageName
// 卸载应用
adb uninstall AppPackageName
// 安装应用
adb install AppPackageName
// 强制停止应用
adb shell am force-stop AppPackageName

// 获取Android版本
adb shell getprop ro.build.version.release
// 获取SDK版本
adb shell getprop ro.build.version.sdk
// 结束ADB调试并关闭
adb shell settings put global adb_enabled 0
// 截图
adb exec-out screencap -p > 文件名.png
